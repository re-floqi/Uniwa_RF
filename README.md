🧠 Υλοποίηση Αντικειμενοστραφούς Προγραμματισμού
java
public class Student {
    // Εγκapsulation με ιδιωτικά πεδία
    private String name;        // Όνομα
    private String studentId;   // Αριθμός Μητρώου
    private String email;       // Φοιτητικό e-mail
    private String department;  // Τμήμα
    private String phone;       // Τηλέφωνο
    
    // Κατασκευαστής
    public Student(String name, String studentId, String email, 
                  String department, String phone) {
        this.name = name;
        this.studentId = studentId;
        this.email = email;
        this.department = department;
        this.phone = phone;
    }
    
    // Getters και Setters
    public String getName() { return name; }
    public void setName(String name) { this.name = name; }
    // ... υπόλοιπες μέθοδοι
}
